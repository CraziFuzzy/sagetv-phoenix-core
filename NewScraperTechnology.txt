var pattern = Pattern.compile("[/\\]([^\(]+)[^0-9a-z]\(([12][0-9]{3})\)");
function GetType() {
   return "tv";
}

function GetName() {
   return "TitleWithYear";
}

function matches(pat, path, group) {
    var matcher = pat.matches(path);
    if (matcher.find()) {
        return matcher.group(group);
    }
}

function GetYear(var path) {
	return matches(pattern, path, 1);
}
